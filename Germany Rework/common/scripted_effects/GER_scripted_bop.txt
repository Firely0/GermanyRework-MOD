GER_army_leader_convert_health = {
	if = {
		limit = {
			has_power_balance_modifier = {
				id = GER_power_balance
				modifier = GER_power_balance_need_wounded_effect
			}
		}
		GER_SUPREM_adolf_hitler = {
			remove_unit_leader_trait = wounded
			add_timed_unit_leader_trait = {
				trait = wounded
				days = 30
			}
			set_unit_leader_flag = {
				flag = recently_wounded
				value = 1
				days = 180
			}
		}
		remove_power_balance_modifier = {
			id = GER_power_balance
			modifier = GER_power_balance_need_wounded_effect
		}
	}
	else_if = {
		limit = {
			has_power_balance_modifier = {
				id = GER_power_balance
				modifier = GER_power_balance_need_sick_effect
			}
		}
		GER_SUPREM_adolf_hitler = {
			remove_unit_leader_trait = sick
			add_timed_unit_leader_trait = {
				trait = sick
				days = 30
			}
			set_unit_leader_flag = {
				flag = recently_sick
				value = 1
				days = 180
			}
		}
		remove_power_balance_modifier = {
			id = GER_power_balance
			modifier = GER_power_balance_need_sick_effect
		}
	}
}
